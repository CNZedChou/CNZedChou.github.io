试除法判定质数

  • AcWing 866. 试除法判定质数

  • 朴素写法：i从2-(n - 1) 进行枚举，判断 n 是否整除 i，时间复杂度为O(n)。

    bool is_prime(int n) {
    // 时间复杂度为O(n)
    if (n < 2) return false;
    for (int i = 2; i < n; i ++) {
        if (n % i == 0) return false;
    }
    return true;
}

  • 根据质数的性质,若在2sqrt(n)中没有可以把它整除的，那么在sqrt(n)n-1中也没有可以把n整除的

    bool is_prime(int n) {
    if (n < 2) return false;
    for (int i = 2; i * i <= n; i ++) {
        if (n % i == 0) return false;
    }
    return true;
}  // 当n 接近 2的32次方时，i * i容易造成溢出
bool is_prime(int n) {
    if (n < 2) return false;
    for (int i = 2; i <= sqrt(n); i ++) {
        if (n % i == 0) return false;
    }
    return true;
}  // 每次进行条件判断的时候都要进行sqrt 计算

  • 改进方法中的最优解

    bool is_prime(int n) {
    if (n < 2) return false;
    for (int i = 2; i <= n / i; i ++) {
        if (n % i == 0) return false;
    }
    return true;
}

筛质数

  • AcWing 868. 筛质数

  • 质数定理：在1-n的数中，有nln⁡n\frac{n}{\ln{n}}lnnn​个质数

  • 朴素筛法：每一个数的所有倍数筛掉，若P没有被筛掉，则可说明从2到(p -1)中，p不是这些数的倍数，故P是质数

    void get_primes(int n) {
    for (int i = 2; i < n; i ++) {
        if (!st[i]) {
            primes[cnt ++] = i; // 如果没有被筛掉则cnt ++
        }
        for (int j = i + i; j <= n; j += i) st[j] = true; // 筛掉所有的i的倍数
    }
}

  • 埃氏筛法：用质数筛，筛掉质数的所有的倍数

    void get_primes2(int n) {
    for (int i = 2; i < n; i ++) {
        if (!st[i]) {
            primes[cnt ++] = i;
            for (int j = i + i; j <= n; j += i) st[j] = true;
        }
    }
}

  • 线性筛法：在埃氏筛法的基础上，让每个合数只被它最小的质因子给筛掉

    void get_primes3(int n) {

AcWing 869. 试除法求约数

  • 约数是成对出现的，枚举从i从1到n，若d能整除n，则n/d 也可以整除n。与试除法求质数类似，只需要枚举i <= n/i 的较小的约数

    vector<int> get_divisors(int n) {
    vector<int> res;
    for (int i = 1; i <= n / i; i ++) {
        if (n % i == 0) {
            res.push_back(i);
            if (i != n / i) res.push_back(n / i);
        }
    }
    sort(res.begin(), res.end());
    return res;
}

int exgcd(int a, int b, int &x, int &y) {
    if (!b) {
        x = 1, y = 0;

int exgcd(int a, int b, int &x, int &y) {
    if (!b) {
        x = 1, y = 0;
        return a;
    }
    int d = exgcd(b, a % b, y, x);
    y -= a / b * x;
    return d;
}
